技術進步所以可以提供類似桌面應用程式的流暢體驗。

各種強大的 JavaScript 雨後春筍般湧出。

模組化

  • Global 
    // jquery.js
// ..
var $ = jQuery;

// app.js
  • AMD: RequireJS 
    // jquery.js
// ..
define('jquery', jQuery);

// app.js
require('jquery', function() { // .. });
  • Common: NodeJS 
    // jquery.js
// ..
module.exports = jQuery;

// app.js
var $ = require('jquery');
  • ES6 
    // jquery.js
// ..
export default jQuery;

// app.js
import $ from 'jquery';
